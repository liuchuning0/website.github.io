         g.drawLine(x1, y1, x2, y2);

public void paintStar(int centerX, int centerY, int outerRadius, Graphics g)
{
   // 设置画笔颜色为黄色
   g.setColor(Color.YELLOW);
   
   // 定义五角星的顶点数量
   final int POINT_COUNT = 5;
   // 创建数组存储顶点坐标
   int[] xPoints = new int[POINT_COUNT * 2]; // 因为五角星需要10个点来定义
   int[] yPoints = new int[POINT_COUNT * 2];
   
   // 计算每个顶点的角度
   double angle = 2 * Math.PI / POINT_COUNT;
   
   // 计算顶点坐标
   for (int i = 0; i < POINT_COUNT; i++)
   {
      // 外圈顶点
      xPoints[i * 2] = (int)(centerX + outerRadius * Math.cos(i * angle - Math.PI / 2));
      yPoints[i * 2] = (int)(centerY + outerRadius * Math.sin(i * angle - Math.PI / 2));
      // 内圈顶点
      xPoints[i * 2 + 1] = (int)(centerX + outerRadius / 2.618 * Math.cos((i + 0.5) * angle - Math.PI / 2));
      yPoints[i * 2 + 1] = (int)(centerY + outerRadius / 2.618 * Math.sin((i + 0.5) * angle - Math.PI / 2));
   }
   
   // 绘制五角星
   g.fillPolygon(xPoints, yPoints, POINT_COUNT * 2);
}

      private void drawFivePointedStar(Graphics2D g2d, int centerX, int centerY, int radius)
      {
         g2d.setColor(Color.YELLOW);
         List<Point> points = new ArrayList<>();
         
         double angleIncrement = Math.PI * 2 / 5;
         for (int i = 0; i < 5; i++)
         {
            double angle = i * angleIncrement + Math.toRadians(18); // Rotate 18 degrees for proper orientation
            double x = centerX + radius * Math.cos(angle);
            double y = centerY + radius * Math.sin(angle);
            points.add(new Point((int) x, (int) y));
            
            double innerRadius = radius * 0.382; // Inner radius for the star points
            double innerAngle = angle + Math.PI / 5;
            double innerX = centerX + innerRadius * Math.cos(innerAngle);
            double innerY = centerY + innerRadius * Math.sin(innerAngle);
            points.add(new Point((int) innerX, (int) innerY));
         }
         
         Polygon star = new Polygon();
         for (Point point : points)
         {
            star.addPoint(point.x, point.y);
         }
         
         g2d.fill(star);
      }
   }

      dataPanel.setLayout(new GridLayout(0, 9));
      
      try (BufferedReader br = new BufferedReader(new FileReader(filePath)))
      {
         String line;
         boolean headerSkipped = false;
         int row = 0;
         while ((line = br.readLine())!= null)
         {
            if (!headerSkipped)
            {
               headerSkipped = true;
               continue;
            }
            row++;
         }
         rowCount = row;
         textFields = new JTextField[rowCount][8];
         comboBoxes = new JComboBox[rowCount];
         
         br.close();
         BufferedReader brForUI = new BufferedReader(new FileReader(filePath));
         headerSkipped = false;
         row = 0;
         while ((line = brForUI.readLine())!= null)
         {
            if (!headerSkipped)
            {
               headerSkipped = true;
               continue;
            }
            String[] columns = line.split("\t");
            if (columns.length!= 9)
            {
               System.err.println("Skipping invalid line: "+ line);
               continue;
            }

            {

               gradeComboBox.setName("cmb"+ row);
               comboBoxes[row] = gradeComboBox;

               System.err.println("Error initializing grade combo box at row "+ row + ": "+ e.getMessage());

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;
public class DataStatistics
{
   public static void main(String[] args)
   {
      if (args.length != 1)
      {
         System.err.println("Usage: java DataStatistics <file-path>");
         return;
      }
      
      Scanner scanner = null;
      try
      {
         scanner = new Scanner(new File(args[0]));
         
         int n; // total rows
         int row = 0, col = 0;
         float sum;
         String grades = "FDCBA";
         String[] headings = new String[9];
         float[][] nums = new float[48][]; // actual n is less than 48
         float[] means = new float[9];
         float[] stdvs = new float[9];
         
         while (scanner.hasNextLine())
         {
            String line = scanner.nextLine();
            String[] columns = line.split("\t");
            if (row == 0)
            {
               headings = columns;
               row++;
               continue;
            }
            
            // Initialize the current row array
            nums[row - 1] = new float[9];
            
            for (col = 0; col < columns.length; col++)
            {
               if (col == 1)
               {
                  String gradeStr = columns[col].trim();
                  nums[row - 1][col] = grades.indexOf(gradeStr);
               }
               else
               {
                  nums[row - 1][col] = Float.parseFloat(columns[col].trim());
               }
            }
            
            row++;
         }
         
         n = row - 1;
         
         // Calculate means
         for (int c = 0; c < 9; c++)
         {
            sum = 0;
            for (int r = 0; r < n; r++)
            {
               sum += nums[r][c];
            }
            means[c] = sum / n;
         }
         
         // Calculate standard deviations
         for (int c = 0; c < 9; c++)
         {
            sum = 0;
            for (int r = 0; r < n; r++)
            {
               sum += Math.pow(nums[r][c] - means[c], 2);
            }
            stdvs[c] = (float) Math.sqrt(sum / n);
         }
         
         // Print results
         System.out.printf("Mean of ID: %.2f\n", means[0]);
         System.out.printf("Mean of Grade: %.2f\n", means[1]);
         System.out.printf("Mean of Score: %.2f\n", means[2]);
         System.out.printf("Mean of Absence: %.2f\n", means[3]);
         System.out.printf("Mean of Time: %.2f\n", means[4]);
         System.out.printf("Mean of Start: %.2f\n", means[5]);
         System.out.printf("Mean of Sub: %.2f\n", means[6]);
         System.out.printf("Mean of Quiz: %.2f\n", means[7]);
         System.out.printf("Mean of Order: %.2f\n", means[8]);
         
         System.out.printf("Stdv of ID: %.2f\n", stdvs[0]);
         System.out.printf("Stdv of Grade: %.2f\n", stdvs[1]);
         System.out.printf("Stdv of Score: %.2f\n", stdvs[2]);
         System.out.printf("Stdv of Absence: %.2f\n", stdvs[3]);
         System.out.printf("Stdv of Time: %.2f\n", stdvs[4]);
         System.out.printf("Stdv of Start: %.2f\n", stdvs[5]);
         System.out.printf("Stdv of Sub: %.2f\n", stdvs[6]);
         System.out.printf("Stdv of Quiz: %.2f\n", stdvs[7]);
         System.out.printf("Stdv of Order: %.2f\n", stdvs[8]);
         
      }
      catch (FileNotFoundException e)
      {
         System.err.println("File not found: "+ e.getMessage());
      }
      finally
      {
         if (scanner != null)
         {
            scanner.close();
         }
      }
   }
}

public class FileSaving
{
   private AtomicBoolean isFileReady = new AtomicBoolean(false);

   private CountDownLatch latch;
   
   static class FileThread extends Thread
   {
      String path;
      private FileSaving fileSaving;
      
      FileThread(String p, FileSaving fileSaving)
      {
         path = p;
         this.fileSaving = fileSaving;
      }
      
      @Override
      public void run()
      {
         String n = Thread.currentThread().getName();
         FileWriter f = null;
         try
         {
            fileSaving.fileWriteLock.lock();
            f = new FileWriter(path, true);
            f.write(n + "\n");
            System.out.println(n + " 线程已将内容写入文件\n");
         }
         catch (IOException e)
         {
            System.out.println("文件写入出现异常: "+ e.toString());
         }
         finally
         {
            if (f!= null)
            {
               try
               {
                  f.close();
               }
               catch (IOException e1)
               {
                  System.out.println("文件关闭出现异常: "+ e1.toString());
               }
            }
            fileSaving.fileWriteLock.unlock();
            if (fileSaving.latch!= null)
            {
               fileSaving.latch.countDown();
            }
         }
      }
   }

import java.util.logging.Level;
import java.util.logging.Logger;

public class GoGameServer
{
   private static final Logger LOGGER = Logger.getLogger(GoGameServer.class.getName());
   private Map<String, Set<Integer>> stones = new HashMap<>();
   private final int N = 19;
   private final int port = 2311;
   private ExecutorService pool = Executors.newFixedThreadPool(10);
   
   class Worker implements Runnable
   {
      private Socket socket;
      
      public Worker(Socket s)
      {
         this.socket = s;
      }
      
      @Override
      public void run()
      {
         try (DataInputStream dis = new DataInputStream(socket.getInputStream()))
         {
            String ip = socket.getInetAddress().getHostAddress();
            synchronized (stones)
            {
               stones.putIfAbsent(ip, new HashSet<>());
               Set<Integer> coordinates = stones.get(ip);
               int r = dis.readInt();
               int c = dis.readInt();
               coordinates.add(r * 100 + c);
               
               double distance = Double.MAX_VALUE;
               int minr = 0, minc = 0;
               for (int row = 0; row < N; row++)
               {
                  for (int col = 0; col < N; col++)
                  {
                     if (coordinates.contains(row * 100 + col))
                     {
                        continue;
                     }
                     double dist = Math.sqrt((col - c) * (col - c) + (row - r) * (row - r));
                     if (dist < distance)
                     {
                        distance = dist;
                        minr = row;
                        minc = col;
                     }
                  }
               }
               coordinates.add(minr * 100 + minc);
            }
            sendNum(ip, minr, minc);
         }
         catch (Exception e)
         {
            LOGGER.log(Level.SEVERE, "Error processing client request", e);
         }
         finally
         {
            try
            {
               socket.close();
            }
            catch (Exception e)
            {
               LOGGER.log(Level.SEVERE, "Error closing socket", e);
            }
         }
      }
   }
   
   protected void sendNum(String ip, int r, int c)
   {
      new Thread(() ->
      {
         try (Socket s = new Socket(ip, port);
         DataOutputStream dos = new DataOutputStream(s.getOutputStream()))
         {
            dos.writeInt(r);
            dos.writeInt(c);
         }
         catch (Exception e)
         {
            LOGGER.log(Level.SEVERE, "Error sending number to client", e);
         }
      }
      ).start();
   }
   
   public void receive()
   {
      try (ServerSocket ss = new ServerSocket(port))
      {
         LOGGER.info("Server started on port "+ port);
         while (true)
         {
            Socket s = ss.accept();
            Worker w = new Worker(s);
            pool.execute(w);
         }
      }

      }

      }
